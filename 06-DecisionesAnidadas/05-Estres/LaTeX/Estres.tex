        let nivelEstres = obtenerNivelEstres(puntajeTotal);
        let reporteEstres = generarReporteEstres(nivelEstres, puntajeTotal);
        mostrarMensaje(reporteEstres);
    }
    function mostrarMensaje(mensaje) {
		this.getField("salida").value = mensaje;
	}
    function mostrarError(mensaje) {
        app.alert(mensaje);
    }
    function ingresarRespuesta(componente) {
        let opcion = this.getField(componente).value;
        if (opcion == 'Off'){
            mostrarError(`Falt\\xf3 una respuesta en la ${componente}, se asume la primera opci\\xf3n`);
            opcion = 1;
        }
        return opcion;
    }
    function ingresarRespuestaInvertida(componente) {
        return Math.abs(ingresarRespuesta(componente) - OPCIONES);
    }
    function calcularPuntajeTotal(respuesta01, respuesta02, respuesta03, respuesta04,
        respuesta05, respuesta06, respuesta07, respuesta08, respuesta09, 
        respuesta10, respuesta11, respuesta12, respuesta13, respuesta14) {
        return respuesta01 + respuesta02 + respuesta03 + respuesta04 +
            respuesta05 + respuesta06 + respuesta07 + respuesta08 +
            respuesta09 + respuesta10 + respuesta11 + respuesta12 +
            respuesta13 + respuesta14;
    }
    function obtenerNivelEstres(puntajeTotal) {
        let nivelEstres;
        if (puntajeTotal < NIVEL_BAJO) {
            nivelEstres = 'BAJO';
        } else {
            if (puntajeTotal < NIVEL_MODERADO) {
                nivelEstres = 'MODERADO';
            } else {
                nivelEstres = 'ALTO';
            }
        }
        return nivelEstres;
    }
    function generarReporteEstres(nivelEstres, puntajeTotal) {
        return `\\nSu nivel de estr\\xe9s es ${nivelEstres}, `+
            `con un puntaje de ${puntajeTotal}.`;
    }
